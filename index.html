// Форматирование с пробелами для тысяч function formatNumber(num) { const rounded = Math.round(num); return rounded.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ' '); } // Форматирование с десятичными (2 знака) и пробелами для тысяч function formatNumberWithDecimals(num) { const rounded = Math.round(num * 100) / 100; const parts = rounded.toString().split('.'); parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ' '); if (parts.length === 1) { return parts[0] + '.00'; } if (parts[1].length === 1) { parts[1] = parts[1] + '0'; } return parts.join('.'); }
